clearclaims.bat

@echo off

set Log=C:\scripts\logs\clearclaims\%date:~-4,4%%date:~-10,2%%date:~-7,2%_report.log

echo Job run on %date% at %time% by %UserName% >> %Log%

echo ** Start scan results ** >> %Log%

C:\Multicraft\mysql\bin\mysql.exe -u -p db < C:\scripts\sql\check.sql 1>>%Log% 2>>&1

C:\Multicraft\mysql\bin\mysql.exe -u -p db < C:\scripts\sql\delete.sql 1>>%Log% 2>>&1

check.sql

SELECT * FROM griefprevention_claimdata

WHERE lessercorner NOT LIKE "world%"

AND lessercorner NOT LIKE "DIM1;%"

AND lessercorner NOT LIKE "DIM-1;%"

AND lessercorner NOT LIKE "DIM6;%"

AND lessercorner NOT LIKE "DIM7;%"

AND lessercorner NOT LIKE "DIM-37;%"

AND lessercorner NOT LIKE "DIM-42;%"

AND lessercorner NOT LIKE "DIM-112;%"
